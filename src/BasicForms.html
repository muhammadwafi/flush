Basic Form
Basic form element styling
Your username must be unique and without space
Your email won't be shown to public
Your password must be unique and 8 letters minimum

By clicking "Create Account" button you are agreeing to the terms of use and acknowledging the privacy policy 

<form>
    <div class="form-group">
        <label for="username">Username</label>
        <input type="text" class="form-control" id="username"
            aria-describedby="usernameHelp" placeholder="e.g johndoe" required="required">
        <small id="usernameHelp" class="form-text text-muted">Your username must be unique and without space</small>
    </div>
    <div class="form-group">
        <label for="email">Email</label>
        <input type="email" class="form-control" id="email"
            aria-describedby="emailHelp" 
            placeholder="e.g johndoe@mail.com"
            required="required">
        <small id="emailHelp" class="form-text text-muted">Your email won't be shown to public</small>
    </div>
    <div class="form-group">
        <label for="password">Password</label>
        <input type="password" class="form-control" id="password"
            aria-describedby="passwordHelp" required="required">
        <small id="passwordHelp" class="form-text text-muted">Your password must be unique and 8 letters minimum</small>
    </div>
    <div class="form-row mt-4">
        <div class="col-md-8">
            <div class="form-group">
                <p class="text-muted small">
                    By clicking "Create Account" button you are agreeing to the terms of use and acknowledging the privacy policy
                </p>
            </div>
        </div>
        <div class="col-md-4">
            <div class="form-group text-right">
                <button type="submit" class="btn btn-primary">Create Account</button>
            </div>
        </div>
    </div>
</form>

Custom Input Addon

Custom input group addon by adding input-group-addon element

<form>
    <div class="form-group">
        <label for="usernameAddon">Username</label>
        <div class="input-group custom">
            <div class="input-group-addon">
                <div class="input-group-text">
                    <i data-feather="user"></i>
                </div>
            </div>
            <input type="text" name="username" class="form-control" placeholder="e.g John Doe" id="usernameAddon">
        </div>
    </div>
    <div class="form-group">
        <label for="passwordAddon">Password</label>
        <div class="input-group custom">
            <div class="input-group-addon">
                <div class="input-group-text">
                    <i data-feather="lock"></i>
                </div>
            </div>
            <input type="password" name="password" class="form-control" id="passwordAddon" placeholder="Secure password">
        </div>
    </div>
</form>
